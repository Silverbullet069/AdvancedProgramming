void Parse (string s)
{
    transform(s.begin(), s.end(), s.begin(), ::toupper);

    {
        for(int i = 0 ; s[i] != '\0' ; ++i)
            cout<<my_font[s[i] - 'A'][l];
        cout<<endl;
    }
}

    Parse(s);
