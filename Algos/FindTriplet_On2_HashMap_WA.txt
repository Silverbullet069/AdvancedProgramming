#include <iostream>
#include <unordered_set>

using namespace std;

void FindTriplet(int arr[], const int& n)
{
    for(int i = 0 ; i < n - 1 ; ++i)
    {
        unordered_set<int> s;
        for(int j = i + 1 ; j < n ; ++j)
        {
            int x = -(arr[i] + arr[j]);
            if(s.find(x) != s.end())
                cout<<x<<" "<<arr[i]<<" "<<arr[j]<<"\n"; 
            else
                s.insert(arr[j]);
        }
    }
}

int Ascending(const void* a, const void* b)
{
	return ( *(int*)a - *(int*)b );
}

int main()
{
   //Input
    int n; cin>>n;
    int arr[n];
    for(int i = 0 ; i < n ; ++i)
        cin>>arr[i];
    
    //Output
    qsort(arr, n, sizeof(arr[0]), Ascending);
    FindTriplet(arr, n);
    
    return 0;
}
